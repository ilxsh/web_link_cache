第六个Verilog程序---readmemh的使用

godspeed 2012-08-10 10:23:55
很多时候，需要把内存的内容从文件读入进行使用。
还好有readmemh函数。

module memory();
reg [7:0] my_mem [0:255];
initial
begin
        /*$readmemh("file",mem_array,start_addr,stop_addr);*/
        $readmemh("mem.list", my_mem);
        $display("0x00: %h", my_mem[8'h00]);
        $display("0x01: %h", my_mem[8'h01]);
        $display("0x55: %h", my_mem[8'h55]);
        $display("0x56: %h", my_mem[8'h56]);
end

endmodule
/*
        //comments are allowed
        0xab //addr 8'h00
        0xba //addr 8'h01
        @55 //jump to 8'h55
        0x55 //addr 8'h55
        0xaa //addr 8'h56
*/
