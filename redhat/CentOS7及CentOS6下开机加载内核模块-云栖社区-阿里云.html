    以ceph模块为例：
    # cd /etc/sysconfig/modules/
    # vim ceph.modules
    在文件中添加如下内容 
    #!/bin/sh 
    /sbin/modinfo -F filename ceph > /dev/null 2>&1 
    if [ $? -eq 0 ]; then 
        /sbin/modprobe ceph 
    fi
    # chmod 755 /etc/sysconfig/modules/ceph.modules
    # reboot
    # lsmod | grep ceph
