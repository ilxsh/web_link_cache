openvpn 批量生成用户脚本 

 1 #/bin/bash
 2 for user in "$@"
 3  do
 4      echo "新增用户:$user"
 5      
 6      if [ -d "/etc/openvpn/client/$user" ]
 7         then rm -rf /etc/openvpn/client/$user
 8         rm -rf  /etc/openvpn/easy-rsa/3.0.3/pki/reqs/$user.req
 9         sed -i '/'"$user"'/d' /etc/openvpn/easy-rsa/3.0.3/pki/index.txt
10      fi
11  
12      echo "生产证书"
13      cd /etc/openvpn/client/easy-rsa/3.0.3
14      ./easyrsa init-pki
15      ./easyrsa gen-req $user nopass
16      
17      echo "签约客户端证书"
18      cd /etc/openvpn/easy-rsa/3.0.3/
19      ./easyrsa import-req /etc/openvpn/client/easy-rsa/3.0.3/pki/reqs/$user.req  $user
20      ./easyrsa sign client $user
21      
22      echo "整理证书"
23      mkdir -p  /etc/openvpn/client/$user/
24      cp /etc/openvpn/easy-rsa/3.0.3/pki/ca.crt /etc/openvpn/client/$user/
25      cp /etc/openvpn/easy-rsa/3.0.3/pki/issued/$user.crt /etc/openvpn/client/$user/
26      cp /etc/openvpn/client/easy-rsa/3.0.3/pki/private/$user.key /etc/openvpn/client/$user/
27      cp /etc/openvpn/client/admin.ovpn /etc/openvpn/client/$user/$user.ovpn
28      sed -i 's/admin/'"$user"'/g' /etc/openvpn/client/$user/$user.ovpn
29      cd /etc/openvpn/client/$user/
30      zip -r $user.zip *
31      mv /etc/openvpn/client/$user/$user.zip ~
32 done
