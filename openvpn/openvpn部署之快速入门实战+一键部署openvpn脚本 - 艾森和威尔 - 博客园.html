openvpn部署之快速入门实战+一键部署openvpn脚本

个人原创禁止转载

软件环境：

    Centos6.9 x64

    openvpn-2.4.3-1.el6.x86_64.rpm

    easy-rsa-2.2.2-1.el6.noarch.rpm    #推荐使用2.2的版本

    pkcs11-helper-1.11-3.el6.x86_64.rpm

网络环境：

    openvpn server：192.168.98.130

    client：192.168.98.134

 

服务端安装步骤：

    # 本次安装采用rpm方式安装无需编译，首先安装openvpn以及相关依赖包

    # rpm -ivh pkcs11-helper-1.11-3.el6.x86_64.rpm

    # rpm -ivh openvpn-2.4.3-1.el6.x86_64.rpm

    # rpm -ivh easy-rsa-2.2.2-1.el6.noarch.rpm

    # 创建并编辑配置文件

    # vim /etc/openvpn/server.conf

        port 1194             #指定监听的本机端口号
        proto udp             #指定采用的传输协议，可以选择tcp或udp
        dev tun               
        ca ca.crt             #指定CA证书的文件路径
        cert server.crt       #指定服务器端的证书文件路径
        key server.key        #指定服务器端的私钥文件路径
        dh dh2048.pem         #指定迪菲赫尔曼参数的文件路径
        server  172.16.2.0 255.255.255.0 #指定客户端网段，此处配置的服务器自身占用172.16.2.1
        push "route 10.10.0.0 255.255.0.0" #指定策略路由，访问10.10.0.0网段时才经过openvpn
        ifconfig-pool-persist ipp.txt #将IP分配信息存储到ipp.txt
        client-to-client
        keepalive 10 120
        comp-lzo
        persist-key
        persist-tun
        status openvpn-status.log
        log-append  openvpn.log #日志存放路径
        verb 3        #日志级别

    # 接下来配置证书

    # mkdir -p /etc/openvpn/easy-rsa/keys

    # cp -a /usr/share/easy-rsa/2.0/* /etc/openvpn/easy-rsa/

    # vim /etc/openvpn/easy-rsa/vars

        export KEY_COUNTRY="CN"       #国家
        export KEY_PROVINCE="BJ"      #省
        export KEY_CITY="Bei Jing"    #市
        export KEY_ORG="test"    #定义所在的组织
        export KEY_EMAIL="eisen@blog.51cto.com"    #邮箱
        export KEY_OU="test"    #定义所在的单位
        export KEY_NAME="vpnserver"    #定义openvpn服务器的名称
        export KEY_CN="test"    这行配置，前边默认的“#”号注释去掉

 # cp /etc/openvpn/easy-rsa/openssl-1.0.0.cnf  /etc/openvpn/easy-rsa/openssl.cnf

    # cd /etc/openvpn/easy-rsa

    # source ./vars

    # ./clean-all

    # ./build-ca    #一路回车即可

    # 创建服务端证书

    # ./build-key-server server    #创建名字为server的服务器证书，注意如下输入用户名和名字 

        A challenge password []:password    

        An optional company name []:vpnserver 

        按两次Y

    # 创建客户端证书

    # ./build-dh     #创建秘钥

    # ./build-key client    #创建名称为client的客户端证书，用户名和密码要和服务端统一

        A challenge password []:password    

        An optional company name []:vpnserver 

        按两次Y

    # 防火墙设置

    # iptables -t nat -F    #清空nat规则链

    # iptables -t nat -X    #删除自定义的空规则链

    # iptables -F            

    # iptables -X    #以上是清空防火墙设置，如果防火墙中有其他规则，请慎用以上设置

    # iptables -P FORWARD ACCEPT

    # iptables -I INPUT -p udp --dport 1194 -j ACCEPT    #放行1194端口

    # vi /etc/sysctl.conf   #编辑配置文件，添加以下配置，设置永久路由转发

        net.ipv4.ip_forward=1

    # sysctl -p

    # 配置NAT

    # iptables -t nat -A POSTROUTING -s 172.16.2.0/24 -j SNAT --to-source 192.168.98.130

    # service iptables save

    # 拷贝服务器证书到相应路径

    # cd /etc/openvpn/easy-rsa/keys

    # cp dh2048.pem ca.crt server.crt server.key /etc/openvpn

    # 客户端所用到的证书

        ca.crt

        client.crt

        client.key

    # 启动openvpn

    # service openvpn start

 

配置windows客户端

    # 安装openvpn https://www.techspot.com/downloads/5182-openvpn.html

    # 将客户端证书保存到openvpn安装目录中config下

    # 在config目录下创建并编辑配置文件sample.ovpn

        client

        dev tun

        proto udp

        remote 192.168.98.130 1194    #服务器地址

        resolv-retry infinite

        nobind

        persist-key

        persist-tun

        ca ca.crt  #客户端证书

        cert client.crt       #客户端证书

        key client.key        #客户端证书

        ns-cert-type server

        comp-lzo

        verb 3

    # 启动openvpn双击右下角openvpn图标将会自动连接，右下角openvpn图标变成绿色说明连接成功

 

配置linux客户端

    # rpm -ivh pkcs11-helper-1.11-3.el6.x86_64.rpm

    # rpm -ivh openvpn-2.4.3-1.el6.x86_64.rpm

    # 将客户端证书文件拷贝到/etc/openvpn下

    # 创建并编辑配置文件client.conf，内容与windows配置文件一致

    # openvpn /etc/openvpn/client.conf &     #启动openvpn客户端

 

openvpn客户端证书的增加

    # 先执行vars在运行客户端证书生成工具

    # cd /etc/openvpn/easy-rsa

    # source ./vars

    # ./build-key client11    #创建名称为client的客户端证书，用户名和密码要和服务端统一

        A challenge password []:password    

        An optional company name []:vpnserver 

        按两次Y
