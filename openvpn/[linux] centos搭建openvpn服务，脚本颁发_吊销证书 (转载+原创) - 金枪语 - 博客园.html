# 安装EPEL源（默认yum源没有openvpn和easy-rsa软件包）
rpm -ivh http://mirrors.ustc.edu.cn/centos/6/extras/x86_64/Packages/epel-release-6-8.noarch.rpm
# 安装openvpn和easy-rsa软件包
yum -y install openvpn easy-rsa
 
# 切换到/usr/share/easy-rsa/2.0/目录
cd /usr/share/easy-rsa/2.0/
# 编辑好vars变量
vim vars
...
export KEY_COUNTRY="CN"
export KEY_PROVINCE="FJ"
export KEY_CITY="PuTian"
export KEY_ORG="GongSiMing"
export KEY_EMAIL="your@email.com"
export KEY_OU="MyOrganizationalUnit"
...
# 初始化环境变量
source vars
# 清除所有与证书相关的文件
./clean-all
# 生成CA相关文件
./build-ca
# 生成服务端相关文件
./build-key-server server
# 生成客户端相关文件
./build-key user1
# 生成dh2048.pem文件
./build-dh
# 生成ta.key文件（防DDos攻击）
openvpn --genkey --secret keys/ta.key
 
# 创建/etc/openvpn/server.conf文件，内容如下
port 1194
proto udp
dev tun
ca /usr/share/easy-rsa/2.0/keys/ca.crt
cert /usr/share/easy-rsa/2.0/keys/server.crt
key /usr/share/easy-rsa/2.0/keys/server.key
dh /usr/share/easy-rsa/2.0/keys/dh2048.pem
tls-auth /usr/share/easy-rsa/2.0/keys/ta.key 0
server 10.8.0.0 255.255.255.0
ifconfig-pool-persist ipp.txt
push "redirect-gateway def1 bypass-dhcp"
push "dhcp-option DNS 223.5.5.5"
keepalive 10 120
comp-lzo
user nobody
group nobody
persist-key
persist-tun
status /var/log/openvpn-status.log
log /var/log/openvpn.log
verb 3
duplicate-cn
 
# 修改/etc/sysctl.conf文件，开启路由转发功能
sed -i '/net.ipv4.ip_forward/s/0/1/' /etc/sysctl.conf
# 使刚才的修改立即生效
sysctl -p
 
# 防火墙配置，让来源地址为10.8.0.0/24网段的客户端请求在服务端经过SNAT将数据包中源地址经过MASQUERADE处理改为服务端公网口地址，并由eth1网卡转发出去。
# eth1为服务器公网出口网卡，10.8.0.0根据上文配置文件为客户端分配的ip地址网段
iptables -t nat -A POSTROUTING -o eth1 -s 10.8.0.0/24 -j MASQUERADE
 
# 启动openvpn服务
service openvpn start
 
# 创建一份客户端文件（命名为user1.ovpn），内容如下
client
dev tun
proto udp
remote 服务端公网IP 1194
resolv-retry infinite
nobind
persist-key
persist-tun
ns-cert-type server
comp-lzo
verb 3
tls-auth [inline] 1
<ca>
将/usr/share/easy-rsa/2.0/keys/ca.crt的全部内容复制粘贴于此
</ca>
<cert>
将/usr/share/easy-rsa/2.0/keys/user1.crt的全部内容复制粘贴于此
</cert>
<key>
将/usr/share/easy-rsa/2.0/keys/user1.key的全部内容复制粘贴于此
</key>
<tls-auth>
将/usr/share/easy-rsa/2.0/keys/ta.key的全部内容复制粘贴于此
</tls-auth>

　　客户端配置：

下载安装openvpn gui(https://openvpn.net/需FQ，翻不了就墙内随便下载一个)
从服务端下载user1.ovpn，并将其复制到openvpn的安装目录下的config目录里，启动openvpn程序，连接服务端，如果能获取到10.8.0.0网段ip，就表示配置成功了。如果不能正常浏览网页，检查服务端路由转发和防火墙nat表。

 

需求：

公司要求临时性给员工开通openvpn，网上搜索并未找到合适的临时关闭员工openvpn权限方法，只能手工每次吊销证书和颁发证书，所以写了一个脚本方便操作

# 切换到/usr/share/easy-rsa/2.0/目录
cd /usr/share/easy-rsa/2.0/
# 初始化环境变量
source vars
 
# 吊销client用户证书
# 这条命令执行完成之后，会在keys目录下面，生成一个crl.pem文件，这个文件中包含了吊销证书的名单。
# 成功注销某个证书之后，可以打开keys/index.txt文件，可以看到被注销的证书前面，已标记为R。
./revoke-full user1
 
# 复制crl.pem到/etc/openvpn目录，每一次吊销都需要重新拷贝一次。
\cp keys/crl.pem /etc/openvpn/
 
# 编辑/etc/openvpn/server.conf添加crl-verify选项
# 把crl.pem放在server.conf相同路径下，如果放在/usr/share/easy-rsa/2.0/keys路径下，openvpn运行用户(上文为nobody)要有keys目录的写权限，就会造成keys目录不安全。
crl-verify crl.pem
 
# 重启OpenVPN服务
service openvpn restart

#!/bin/bash
PATH=/usr/local/sbin:/usr/local/bin:/sbin:/bin:/usr/sbin:/usr/bin
action=$1
vpn_user=$2
vpn_ip=服务器ip
ovpn_file=/tmp/$vpn_user.ovpn
expect_script=/data/script/vpn_expect
 
# 脚本用法
usage(){
    echo "Usage: $0 {build|revoke} username"
    exit 1
}
 
# 制作客户端证书
build_ovpn(){
    cat > $ovpn_file <<- EOF
    client
    dev tun
    proto udp
    remote $vpn_ip 1194
    resolv-retry infinite
    nobind
    persist-key
    persist-tun
    ns-cert-type server
    comp-lzo
    verb 3
    tls-auth [inline] 1
    EOF
 
    echo "<ca>" >> $ovpn_file
    cat keys/ca.crt >> $ovpn_file
    echo "</ca>" >> $ovpn_file
     
    echo "<cert>" >> $ovpn_file
    sed -n '/-----BEGIN/,/-----END/p' keys/$vpn_user.crt >> $ovpn_file
    echo -e "</cert>" >> $ovpn_file
     
    echo "<key>" >> $ovpn_file
    cat keys/$vpn_user.key >> $ovpn_file
    echo "</key>" >> $ovpn_file
     
    echo "<tls-auth>" >> $ovpn_file
    cat keys/ta.key >> $ovpn_file
    echo "</tls-auth>" >> $ovpn_file
}
 
# 判断脚本传参个数
[ "$#" -ne 2 ] && {
    echo "Wrong number of argvs,it must be two."
    usage
}
 
case $action in
    build)
        # 检测vpn环境
        cd /usr/share/easy-rsa/2.0
        source vars &> /dev/null
        [[ -f keys/$vpn_user.crt || -f keys/$vpn_user.key ]] && {
            echo "User $vpn_user already exist."
            exit 1
        }
        [[ ! -f keys/ca.crt || ! -f keys/server.key || ! -f keys/dh2048.pem || ! -f keys/ta.key ]] && {
            echo "File keys/ca.crt or keys/ta.key not found."
            echo "Pls run ./build-ca and ./build-key-server server and ./build-dh and openvpn --genkey --secret keys/ta.key first."
            exit 1
        }
 
        # 检查是否已安装expect
        rpm -q expect &> /dev/null || {
            echo "Pls install expect first."
            exit 1
        }
 
        [ ! -f $expect_script ] && {
            echo "Script:$expect_script not found."
            exit 1
        }
         
        # 开始制作证书
        $expect_script $vpn_user &> /dev/null
        build_ovpn
        echo "Build $vpn_user key success."
        echo "Download the file $ovpn_file to your client computer"
        ;;
    revoke)
        # 吊销证书
        cd /usr/share/easy-rsa/2.0
        source ./vars &> /dev/null
        ./revoke-full $vpn_user
        \cp keys/crl.pem /etc/openvpn/
        rm keys/$vpn_user* -f
        rm $ovpn_file -f
        ;;
    *)
        echo "Undefine action"
        usage
        ;;
esac
